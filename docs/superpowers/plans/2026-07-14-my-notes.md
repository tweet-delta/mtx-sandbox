# "My notes" Personal Checklist Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give each tech a private, cloud-synced personal checklist (📋 My notes) reachable from the home screen — add items anytime, check them off (they stay crossed out until cleared), fully isolated per tech with no supervisor access.

**Architecture:** New Supabase table `personal_notes`, RLS-scoped to `tech_id = auth.uid()` with no supervisor exception (unlike `daily_logs`). Five new `cloud.js` functions mirroring the Daily Logs manual-note CRUD shape. New `#mynotes` hash-router screen in `index.html`, following the exact pattern already used by `#profile` / `#history` / `#logs`.

**Tech Stack:** Vanilla HTML/CSS/JS (no build step), Supabase Postgres + RLS via the CLI (`supabase db push`), existing hash-router + screen-visibility CSS pattern.

## Global Constraints

- Copied verbatim from spec: **"No supervisor exception"** — every RLS policy on `personal_notes` scopes strictly to `tech_id = auth.uid()`. This is the one place in the app where even supervisors get zero read access to another user's rows.
- Copied verbatim from spec: **"Adding a new item never affects existing ones; no cap on list length."**
- Copied verbatim from spec: **"Checked items stay visible, crossed out, until manually cleared"** — checking an item must never delete or reorder it.
- All client-side mutators are self-scoped (`tech_id = me`) as defense-in-depth atop RLS, matching every other `cloud.js` mutator in this codebase (see `addLogEntry`/`updateLogEntry`/`deleteLogEntry` in `cloud.js` for the existing pattern).
- No automated test harness exists in this repo (per `CLAUDE.md`). "Verify" means driving the app in a real browser end-to-end, plus a headless parse check (zero console errors) after markup/JS edits.
- **Never run destructive remote Supabase commands.** Migrations are applied with `supabase db push --workdir "c:\Big Dogs Apps\MTX Checklist V1"` only.
- Service worker: bump `CACHE` in `route-checklist/sw.js` to the next version after `index.html`/`cloud.js` change (currently `route-checklist-v21` → this slice ships `v22`). Remind the owner to hard-refresh (Ctrl+Shift+R) and fully close/reopen the PWA.

```sql
-- ============================================================================
-- MTX Route Checklist — My notes (private personal checklist)
-- A per-tech scratchpad: shopping lists, reminders, "bring tomorrow" items.
-- Fully private — unlike daily_logs, there is NO supervisor read exception.
-- Spec: docs/superpowers/specs/2026-07-14-my-notes-design.md
-- ============================================================================

create table if not exists public.personal_notes (
  id         uuid primary key default gen_random_uuid(),
  tech_id    uuid not null references public.profiles (id) on delete cascade
               default auth.uid(),
  text       text not null,
  done       boolean not null default false,
  position   int not null default 0,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

-- The list screen queries by tech, ordered by insertion position.
create index if not exists personal_notes_tech_position_idx
  on public.personal_notes (tech_id, position);

alter table public.personal_notes enable row level security;

-- Every policy is scoped to tech_id = auth.uid() with NO supervisor
-- exception — this table is a personal scratchpad, not work history.
create policy personal_notes_select on public.personal_notes
  for select using (tech_id = auth.uid());

create policy personal_notes_insert on public.personal_notes
  for insert with check (tech_id = auth.uid());

create policy personal_notes_update on public.personal_notes
  for update using (tech_id = auth.uid()) with check (tech_id = auth.uid());

create policy personal_notes_delete on public.personal_notes
  for delete using (tech_id = auth.uid());
```

```javascript
// ---- My notes (private personal checklist) ----
// Fully tech-scoped, no supervisor read path — see 0018_personal_notes.sql.

async function listMyNotes() {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return [];
  const { data, error } = await supabase
    .from("personal_notes")
    .select("id, text, done, position")
    .eq("tech_id", user.id)
    .order("position", { ascending: true });
  if (error) {
    if (!isMissingTable(error)) console.error("Could not load my notes:", error.message);
    return [];
  }
  return data || [];
}

async function addMyNote(text) {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not signed in." };
  const trimmed = (text || "").trim();
  if (!trimmed) return { error: "Note text can't be empty." };
  const { data: maxRow } = await supabase
    .from("personal_notes")
    .select("position")
    .eq("tech_id", user.id)
    .order("position", { ascending: false })
    .limit(1)
    .maybeSingle();
  const nextPosition = maxRow ? maxRow.position + 1 : 0;
  const { error } = await supabase
    .from("personal_notes")
    .insert({ tech_id: user.id, text: trimmed, position: nextPosition });
  return { error: error ? error.message : null };
}

async function toggleMyNote(id, done) {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not signed in." };
  const { error } = await supabase
    .from("personal_notes")
    .update({ done })
    .eq("id", id)
    .eq("tech_id", user.id);
  return { error: error ? error.message : null };
}

async function deleteMyNote(id) {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not signed in." };
  const { error } = await supabase
    .from("personal_notes")
    .delete()
    .eq("id", id)
    .eq("tech_id", user.id);
  return { error: error ? error.message : null };
}

async function clearCheckedNotes() {
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { error: "Not signed in." };
  const { error } = await supabase
    .from("personal_notes")
    .delete()
    .eq("tech_id", user.id)
    .eq("done", true);
  return { error: error ? error.message : null };
}
```

- [ ] **Step 7: Add `renderMyNotesScreen()` and its event wiring**

Add this block near `renderProfileScreen()` (e.g. directly after it):

```javascript
  // ---- My notes screen (private personal checklist) ----
  async function renderMyNotesScreen() {
    const body = document.getElementById("mynotesBody");
    body.innerHTML = `<p class="screen-sub">Loading…</p>`;
    const items = await window.cloud.listMyNotes();
    if (currentScreenFromHash() !== "mynotes") return;   // navigated away meanwhile
    renderMyNotesList(body, items);
  }

  function renderMyNotesList(body, items) {
    const hasChecked = items.some(i => i.done);
    const rows = items.length
      ? items.map(i => `
        <div class="mynotes-item${i.done ? " done" : ""}">
          <input type="checkbox" data-mynote-toggle="${escAttr(i.id)}" ${i.done ? "checked" : ""} aria-label="Mark done">
          <span>${escHtml(i.text)}</span>
          <button type="button" data-mynote-delete="${escAttr(i.id)}" aria-label="Delete">✕</button>
        </div>`).join("")
      : `<p class="screen-sub">No notes yet — add one above.</p>`;
    body.innerHTML = `
      <div class="mynotes-add">
        <input type="text" id="mynoteInput" placeholder="Add an item…" aria-label="New note text">
        <button type="button" id="mynoteAddBtn">+ Add</button>
      </div>
      <div id="mynotesList">${rows}</div>
      ${hasChecked ? `<button type="button" class="mynotes-clear" id="mynotesClearBtn">Clear checked</button>` : ""}`;
  }

  async function submitMyNote() {
    const input = document.getElementById("mynoteInput");
    const text = input.value;
    if (!text.trim()) return;
    input.disabled = true;
    const res = await window.cloud.addMyNote(text);
    input.disabled = false;
    if (res.error) { toast("Couldn't add — " + res.error, "error"); return; }
    input.value = "";
    renderMyNotesScreen();
  }

  document.getElementById("mynotesBody").addEventListener("click", async (e) => {
    if (e.target.id === "mynoteAddBtn") { submitMyNote(); return; }
    if (e.target.id === "mynotesClearBtn") {
      const res = await window.cloud.clearCheckedNotes();
      if (res.error) { toast("Couldn't clear — " + res.error, "error"); return; }
      renderMyNotesScreen();
      return;
    }
    const delBtn = e.target.closest("[data-mynote-delete]");
    if (delBtn) {
      const res = await window.cloud.deleteMyNote(delBtn.dataset.mynoteDelete);
      if (res.error) { toast("Couldn't delete — " + res.error, "error"); return; }
      renderMyNotesScreen();
      return;
    }
  });
  document.getElementById("mynotesBody").addEventListener("keydown", (e) => {
    if (e.target.id === "mynoteInput" && e.key === "Enter") submitMyNote();
  });
  document.getElementById("mynotesBody").addEventListener("change", async (e) => {
    const cb = e.target.closest("[data-mynote-toggle]");
    if (!cb) return;
    const res = await window.cloud.toggleMyNote(cb.dataset.mynoteToggle, cb.checked);
    if (res.error) { toast("Couldn't save — " + res.error, "error"); cb.checked = !cb.checked; return; }
    renderMyNotesScreen();
  });
```

The checkbox toggle re-renders the whole list after a successful save (rather
than just toggling a class locally) so the "Clear checked" button's
show/hide state and the crossed-out styling always match the server's view of
`done` — consistent with how `renderProfileScreen`/`renderHistoryScreen`
re-fetch rather than patch the DOM piecemeal.
